# Musica Auth System - Code Reference Guide\n\n## 1. Redux State (store/slices/authSlice.js)\n\n### State Shape\n```javascript\n{\n  auth: {\n    user: null | {\n      id: string,\n      email: string,\n      displayName: string,\n      createdAt: string\n    },\n    token: null | string,\n    isAuthenticated: false | true,\n    isLoading: false | true,\n    isHydrated: false | true,\n    error: null | string,\n    successMessage: null | string\n  }\n}\n```\n\n### Async Thunks\n\n#### loginAsync\n```javascript\n// Arguments\n{ email: string, password: string }\n\n// Payload (success)\n{ user: {...}, token: string }\n\n// Error (rejected)\nstring (error message)\n\n// Side effects\n- 1 second delay (mock)\n- Validates email format\n- Validates password length >= 6\n- Returns mock user with JWT token\n```\n\n#### signUpAsync\n```javascript\n// Arguments\n{ email: string, password: string, displayName: string }\n\n// Payload (success)\n{ user: {...}, token: string }\n\n// Error (rejected)\nstring (error message)\n\n// Side effects\n- 1 second delay (mock)\n- Validates all fields present\n- Validates email format (regex)\n- Validates password length >= 6\n- Creates new user with UUID\n```\n\n### Reducers\n\n```javascript\n// logout() - Clears all auth data\nlogout: (state) => {\n  state.user = null;\n  state.token = null;\n  state.isAuthenticated = false;\n  state.error = null;\n  state.successMessage = null;\n}\n\n// clearError() - Removes error message\nclearError: (state) => {\n  state.error = null;\n}\n\n// clearSuccess() - Removes success message\nclearSuccess: (state) => {\n  state.successMessage = null;\n}\n\n// setUserFromStorage() - Hydrates from AsyncStorage\nsetUserFromStorage: (state, action) => {\n  state.user = action.payload.user;\n  state.token = action.payload.token;\n  state.isAuthenticated = !!action.payload.token;\n  state.isHydrated = true;\n}\n\n// setHydrated() - Marks app as ready\nsetHydrated: (state) => {\n  state.isHydrated = true;\n}\n```\n\n## 2. Context API (context/AuthContext.js)\n\n### useAuth() Hook\n\n```javascript\nconst {\n  // State values\n  user,              // User object or null\n  token,             // JWT token or null\n  isAuthenticated,   // Boolean\n  isLoading,         // Boolean (during auth request)\n  isHydrated,        // Boolean (initial load complete)\n  error,             // Error message or null\n  successMessage,    // Success message or null\n  \n  // Methods\n  login,             // async (email, password) => Promise\n  signUp,            // async (email, password, displayName) => Promise\n  logout,            // async () => Promise\n  clearError,        // () => void\n  clearSuccess,      // () => void\n} = useAuth();\n```\n\n### Methods\n\n#### login(email, password)\n```javascript\ntry {\n  const result = await login('user@email.com', 'password123');\n  \n  if (result.meta.requestStatus === 'fulfilled') {\n    console.log('Login successful!');\n    // Navigation happens automatically in RootNavigator\n  } else {\n    console.log('Login failed:', result.payload);\n  }\n} catch (error) {\n  console.error('Login error:', error);\n}\n```\n\n#### signUp(email, password, displayName)\n```javascript\nconst result = await signUp(\n  'newuser@email.com',\n  'password123',\n  'John Doe'\n);\n\nif (result.meta.requestStatus === 'fulfilled') {\n  // User auto-logged in\n}\n```\n\n#### logout()\n```javascript\nawait logout();\n// AsyncStorage cleared\n// ReduxState reset\n// Navigation returns to AuthStack\n```\n\n## 3. Navigation Guard (App.js)\n\n### Conditional Navigation\n\n```javascript\nfunction RootNavigator() {\n  const { isAuthenticated, isHydrated } = useAuth();\n  \n  // Restore from storage - show nothing while loading\n  if (!isHydrated) {\n    return null;  // Splash screen option\n  }\n  \n  return (\n    <NavigationContainer>\n      {isAuthenticated ? (\n        // User logged in - show main app\n        <MainApp />  // Drawer + Tabs + Screens\n      ) : (\n        // User not logged in - show auth screens\n        <AuthStack />  // Login + SignUp\n      )}\n    </NavigationContainer>\n  );\n}\n\nfunction App() {\n  return (\n    <Provider store={store}>\n      <AuthProvider>\n        <RootNavigator />\n      </AuthProvider>\n    </Provider>\n  );\n}\n```\n\n## 5. Form Validation Examples\n\n### Email Validation\n```javascript\nconst validateEmail = (email) => {\n  const regex = /^[^\\s@]+@[^\\s@]+\\.[^\\s@]+$/;\n  return regex.test(email);\n};\n\n// Valid: test@example.com\n// Invalid: test@example, test@, @example.com\n```\n\n### Password Strength\n```javascript\nconst getPasswordStrength = (password) => {\n  if (!password) return { level: 0, text: '', color: '#999' };\n  if (password.length < 6) return { level: 1, text: 'Weak', color: '#e74c3c' };\n  if (password.length < 10) return { level: 2, text: 'Fair', color: '#f39c12' };\n  return { level: 3, text: 'Strong', color: '#1DB954' };\n};\n```\n\n### Form Reducer\n```javascript\nconst formReducer = (state, action) => {\n  switch (action.type) {\n    case 'SET_EMAIL':\n      return { \n        ...state, \n        email: action.payload, \n        emailError: ''  // Clear error on change\n      };\n    \n    case 'SET_ERRORS':\n      return {\n        ...state,\n        emailError: action.payload.emailError ?? state.emailError,\n        passwordError: action.payload.passwordError ?? state.passwordError,\n      };\n    \n    case 'RESET':\n      return {\n        email: '',\n        password: '',\n        emailError: '',\n        passwordError: '',\n      };\n    \n    default:\n      return state;\n  }\n};\n```\n\n## 6. AsyncStorage Integration\n\n### Storage Keys\n```javascript\nconst AUTH_STORAGE_KEY = '@musica_auth';      // { user, token }\nconst AUTH_EXPIRY_KEY = '@musica_auth_expiry';  // timestamp\n```\n\n### Persist Auth\n```javascript\nconst persistAuth = async (user, token) => {\n  const expiryTime = new Date().getTime() + (7 * 24 * 60 * 60 * 1000);\n  \n  await AsyncStorage.setItem(\n    AUTH_STORAGE_KEY,\n    JSON.stringify({ user, token })\n  );\n  \n  await AsyncStorage.setItem(\n    AUTH_EXPIRY_KEY,\n    expiryTime.toString()\n  );\n};\n```\n\n### Restore Auth\n```javascript\nconst loadStoredAuth = async () => {\n  const stored = await AsyncStorage.getItem(AUTH_STORAGE_KEY);\n  const expiry = await AsyncStorage.getItem(AUTH_EXPIRY_KEY);\n  \n  if (stored && expiry && new Date().getTime() < parseInt(expiry)) {\n    const { user, token } = JSON.parse(stored);\n    dispatch(setUserFromStorage({ user, token }));\n  } else {\n    // Expired - clear storage\n    await AsyncStorage.removeItem(AUTH_STORAGE_KEY);\n    dispatch(setHydrated());\n  }\n};\n```\n\n## 7.
